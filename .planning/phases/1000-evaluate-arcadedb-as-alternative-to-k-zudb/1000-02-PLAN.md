function generateLargeCodeGraph(moduleCount: number): CodeGraph {
  const modules: Module[] = [];
  const symbols: Symbol[] = [];

  for (let i = 0; i < moduleCount; i++) {
    const moduleId = `module-${i}`;
    modules.push({
      id: moduleId,
      path: `src/module${i}.ts`,
      language: 'typescript',
      exports: [`export${i}`],
      imports: i > 0 ? [`module-${i - 1}`] : [],
      symbols: [],
      dependencies: [],
      loc: { start: { line: 1, column: 0 }, end: { line: 100, column: 0 } },
    });
  }

  return {
    project: {
      id: 'benchmark-project',
      name: 'Benchmark Project',
      rootPath: '/tmp/benchmark',
      version: '1.0.0',
    },
    modules,
    symbols,

}

async function measureTime<T>(fn: () => Promise<T>): Promise<{ result: T; time: number }> {
  const start = performance.now();
  const result = await fn();
  const time = performance.now() - start;
  return { result, time };
}

function recordResult(name: string, arcadeTime: number, fsTime: number): BenchmarkResult {
  const result = {
    name,
    arcadeDBTime: arcadeTime,
    filesystemTime: fsTime,
    speedup: fsTime / arcadeTime,
  };
  results.push(result);
  return result;
}

describe('ArcadeDB fallback mechanism', () => {
  it('should fallback to filesystem when arcadedb is explicitly requested but unavailable', async () => {
    const warnSpy = vi.spyOn(console, 'warn').mockImplementation(() => {});

    try {
      const storage = await factory.create({ type: 'arcadedb' });

      // 如果 ArcadeDB 不可用，应该 fallback 到 filesystem
      // 如果可用，应该返回 arcadedb
      expect(['arcadedb', 'filesystem']).toContain(storage.type);
    } finally {
      warnSpy.mockRestore();
    }
  });

  it('should log warning when falling back from arcadedb to filesystem', async () => {
    const warnSpy = vi.spyOn(console, 'warn').mockImplementation(() => {});

    try {
      await factory.create({ type: 'arcadedb' });

      const fallbackWarning = warnSpy.mock.calls.find(
        call => typeof call[0] === 'string' && call[0].includes('ArcadeDB unavailable')
      );

      if (fallbackWarning) {
        expect(fallbackWarning[0]).toContain('ArcadeDB unavailable');
        expect(fallbackWarning[0]).toContain('falling back');
      }
    } finally {
      warnSpy.mockRestore();
    }
  });

  it('should report arcadedb availability correctly', () => {
    const isAvailable = StorageFactory.isStorageTypeAvailable('arcadedb');
    expect(typeof isAvailable).toBe('boolean');
  });

  it('should include arcadedb in available types when installed', () => {
    const availableTypes = StorageFactory.getAvailableStorageTypes();
    expect(availableTypes).toContain('arcadedb');
  });
});
