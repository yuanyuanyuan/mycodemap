```sh
# 5a. 轻量真实验证检查（warn-only）
# 如果新增/修改了测试文件，检查是否包含真实调用模式
echo "Checking for real-world test patterns..."
STAGED_TEST_FILES=$(git diff --cached --name-only --diff-filter=ACM | grep '\.test\.ts$' || true)
if [ -n "$STAGED_TEST_FILES" ]; then
    MISSING_REAL_PATTERN=0
    for file in $STAGED_TEST_FILES; do
        if [ ! -f "$file" ]; then
            continue
        fi
        # 检查是否包含真实调用模式
        # 注意：StdioServerTransport + PassThrough 是真实 MCP transport 测试模式（参见 RESEARCH.md Pitfall 1 纠正）
        if ! grep -qE '(spawn|execSync|execFileSync|mktemp|StdioServerTransport|PassThrough|createServer|listen)' "$file"; then
            echo "WARNING: $file does not appear to contain real-world invocation patterns (spawn, execSync, mktemp, StdioServerTransport, PassThrough, etc.)"
            MISSING_REAL_PATTERN=$((MISSING_REAL_PATTERN + 1))
        fi
    done
    if [ $MISSING_REAL_PATTERN -gt 0 ]; then
        echo ""
        echo "Consider adding integration/E2E tests with real subprocess/transport calls."
        echo "See AGENTS.md Section 8.1 for the real-world validation threshold."
        echo "This is a warning only; commit will proceed."
        echo ""
    fi
fi
```

```yaml
      - name: Verify real-world evidence in commits
        if: github.event_name == 'pull_request'
        run: |
          # 检查 PR 的 commit messages 中是否有 [证据] 标签
          # 仅对功能/修复类 commit 要求（非 DOCS/REFACTOR/CONFIG）
          COMMITS=$(git log --format='%s' "${{ github.event.pull_request.base.sha }}..HEAD")
          REQUIRE_EVIDENCE=false
          while IFS= read -r line; do
            if echo "$line" | grep -qE '^\[(FEATURE|BUGFIX)\]'; then
              REQUIRE_EVIDENCE=true
            fi
          done <<< "$COMMITS"
          
          if [ "$REQUIRE_EVIDENCE" = "true" ]; then
            if ! echo "$COMMITS" | grep -q '\[证据\]'; then
              echo "WARNING: FEATURE/BUGFIX commits detected but no [证据] tag found in commit messages."
              echo "Please include evidence of real-world validation in your commit messages or PR description."
              echo ""
              echo "Expected format: [证据] path:line or [证据] command output"
              echo "See docs/rules/testing.md for evidence requirements and AGENTS.md Section 8.1 for threshold."
              echo ""
              # Warn-only for now; can be made blocking after team adoption
              exit 0
            fi
          fi
          echo "Real-world evidence check passed"
```

设计要点（per D-03）：
- 仅对 FEATURE/BUGFIX commit 类型触发
- Warn-only 初始阶段（exit 0），可渐进升级为 blocking
- 引用 AGENTS.md Section 8.1 和 testing.md 作为规则依据
