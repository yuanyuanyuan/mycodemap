# Phase 45 Plan 02: Auto-Remediation Engine & WASM Fallback Handler

## Objective

Implement the `--apply-suggestion` safety-gated auto-remediation engine that evaluates confidence, executes `nextCommand` when threshold is met, logs results to stderr NDJSON, and returns structured `{type: "result", success, data?}` feedback. Implement the `--wasm-fallback` reactive handler that intercepts native dependency failures and switches to WASM loading paths.

## must_haves

- [ ] `tryApplySuggestion` function gates on confidence >= 0.8 && nextCommand present && --apply-suggestion flag
- [ ] Executed suggestion results return `{type: "result", success, data?}` JSON to stderr
- [ ] Auto-exec failure stops immediately and returns new ActionableError (no retry)
- [ ] `handleWasmFallback` reactive handler switches tree-sitter/better-sqlite3 to WASM on native failure
- [ ] stderr NDJSON logging for all apply-suggestion operations

## truths

- Confidence threshold 0.8 is the hard gate — below this, --apply-suggestion is ignored
- No retry on auto-exec failure — prevents cascading damage
- --wasm-fallback is reactive (triggered at error handler level), not proactive (not checked at command entry)
- stderr NDJSON format matches Phase 44 progress event pattern: `{type: "...", ...}`

```typescript
export async function tryApplySuggestion(
  error: ActionableError,
  options: { applySuggestion?: boolean; wasmFallback?: boolean },
  mode: OutputMode
): Promise<SuggestionResult> {
  // Gate 1: --apply-suggestion flag must be set
  if (!options.applySuggestion) {
    return { type: 'result', success: false, data: { reason: 'apply-suggestion flag not set' } };
  }

  // Gate 2: Confidence must meet threshold
  if (error.confidence < APPLY_SUGGESTION_CONFIDENCE_THRESHOLD) {
    return { type: 'result', success: false, data: { reason: `confidence ${error.confidence} below threshold ${APPLY_SUGGESTION_CONFIDENCE_THRESHOLD}` } };
  }

  // Gate 3: nextCommand must exist
  if (!error.nextCommand) {
    return { type: 'result', success: false, data: { reason: 'no nextCommand available' } };
  }

  // Log attempt to stderr NDJSON
  const attemptLog = JSON.stringify({ type: 'apply-suggestion', action: 'attempt', command: error.nextCommand, confidence: error.confidence });
  process.stderr.write(attemptLog + '\n');

  try {
    // Execute the nextCommand as a child process
    const result = await executeNextCommand(error.nextCommand);

    // Log success to stderr NDJSON
    const successLog = JSON.stringify({ type: 'apply-suggestion', action: 'success', command: error.nextCommand });
    process.stderr.write(successLog + '\n');

    return { type: 'result', success: true, data: result, attemptedCommand: error.nextCommand };
  } catch (execError) {
    // NO RETRY — stop immediately on failure
    const failureError = execError instanceof Error ? execError : new Error(String(execError));

    // Log failure to stderr NDJSON
    const failLog = JSON.stringify({ type: 'apply-suggestion', action: 'failed', command: error.nextCommand, error: failureError.message });
    process.stderr.write(failLog + '\n');

    // Return failure result (do NOT throw — caller decides what to do)
    return { type: 'result', success: false, data: { error: failureError.message }, attemptedCommand: error.nextCommand };
  }
}
```

3. Implement `executeNextCommand` — parses the `nextCommand` string and executes via `child_process.exec`:

```typescript
async function executeNextCommand(command: string): Promise<unknown> {
  const { exec } = await import('node:child_process');
  const { promisify } = await import('node:util');
  const execAsync = promisify(exec);

  const { stdout } = await execAsync(command, { timeout: 30000 });
  return { stdout: stdout.trim() };
}
```

```typescript
export async function checkAndActivateWasmFallback(
  module: 'tree-sitter' | 'better-sqlite3',
  options: { wasmFallback?: boolean }
): Promise<WasmFallbackResult> {
  // If --wasm-fallback is not set, just report native status
  if (!options.wasmFallback) {
    return {
      nativeAvailable: await checkNativeModule(module),
      wasmAvailable: false,
      fallbackActivated: false,
      module,
      message: '--wasm-fallback not requested',
    };
  }

  // Try native first
  const nativeAvailable = await checkNativeModule(module);
  if (nativeAvailable) {
    return {
      nativeAvailable: true,
      wasmAvailable: true,
      fallbackActivated: false,
      module,
      message: 'Native module available, WASM fallback not needed',
    };
  }

  // Native failed, try WASM
  const wasmAvailable = await checkWasmModule(module);
  if (wasmAvailable) {
    // Set environment flag so downstream code uses WASM path
    process.env[`CODEMAP_USE_WASM_${module.replace('-', '_').toUpperCase()}`] = '1';

    return {
      nativeAvailable: false,
      wasmAvailable: true,
      fallbackActivated: true,
      module,
      message: `Native ${module} unavailable — activated WASM fallback`,
    };
  }

  return {
    nativeAvailable: false,
    wasmAvailable: false,
    fallbackActivated: false,
    module,
    message: `Both native and WASM ${module} unavailable — install build tools or check WASM module`,
  };
}
```

3. Implement `checkNativeModule` that attempts to dynamically import the native module:
```typescript
async function checkNativeModule(module: 'tree-sitter' | 'better-sqlite3'): Promise<boolean> {
  try {
    if (module === 'tree-sitter') {
      const ts = await import('tree-sitter');
      return !!ts.default || !!ts;
    }
    const sqlite = await import('better-sqlite3');
    return !!sqlite.default || !!sqlite;
  } catch {
    return false;
  }
}
```

4. Implement `checkWasmModule` that checks for WASM alternatives:
```typescript
async function checkWasmModule(module: 'tree-sitter' | 'better-sqlite3'): Promise<boolean> {
  try {
    if (module === 'tree-sitter') {
      await import('web-tree-sitter');
      return true;
    }
    // better-sqlite3 WASM: check for wa-sqlite or sql.js as fallback
    // For now, check if Node.js 22+ built-in sqlite is available
    const nodeVersion = process.versions.node.split('.').map(Number);
    if (nodeVersion[0] >= 22) {
      await import('node:sqlite');
      return true;
    }
    // Try sql.js as WASM fallback
    await import('sql.js');
    return true;
  } catch {
    return false;
  }
}
```

5. Implement `createNativeDepError` helper that creates a standardized ActionableError for native dep failures:

```typescript
export function createNativeDepError(
  module: 'tree-sitter' | 'better-sqlite3',
  nativeError: Error,
  wasmAvailable: boolean
): ActionableError {
  const suggestions: string[] = ['--wasm-fallback'];
  if (wasmAvailable) {
    suggestions.push('WASM module detected — add --wasm-fallback to activate');
  }
  suggestions.push('npm rebuild (rebuild native modules)');
  suggestions.push('Install prebuilt binaries: https://github.com/...' );

  return createActionableError(
    wasmAvailable ? ErrorCodes.DEP_WASM_FALLBACK_AVAILABLE : ErrorCodes.DEP_NATIVE_MISSING,
    `${module} cannot be loaded: ${nativeError.message}`,
    `loading ${module}`,
    {
      rootCause: nativeError.message,
      remediationPlan: suggestions.join('; '),
      confidence: wasmAvailable ? 0.95 : 0.9,
      nextCommand: 'codemap --wasm-fallback',
    }
  );
}
```
