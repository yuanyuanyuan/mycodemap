       async loadCodeGraph(): Promise<CodeGraph> {
         this.ensureInitialized();

         // 加载项目元数据
         const metaRows = await this.queryRows(`
           MATCH (meta:Metadata)
           RETURN meta.projectId as projectId, meta.projectName as projectName,
                  meta.rootPath as rootPath, meta.createdAt as createdAt, meta.updatedAt as updatedAt
           LIMIT 1
         `);

         const meta = metaRows[0];

         // 加载所有模块
         const moduleRows = await this.queryRows(`
           MATCH (m:Module)
           RETURN m.id as id, m.projectId as projectId, m.path as path, m.language as language,
                  m.lines as lines, m.codeLines as codeLines, m.commentLines as commentLines, m.blankLines as blankLines
         `);

         const modules: Module[] = moduleRows.map(row => ({
           id: String(row.id),
           projectId: String(row.projectId),
           path: String(row.path),
           language: String(row.language),
           stats: {
             lines: Number(row.lines),
             codeLines: Number(row.codeLines),
             commentLines: Number(row.commentLines),
             blankLines: Number(row.blankLines),
           },
         }));

         // 加载所有符号
         const symbolRows = await this.queryRows(`
           MATCH (s:Symbol)
           RETURN s.id as id, s.moduleId as moduleId, s.name as name, s.kind as kind,
                  s.visibility as visibility, s.file as file, s.line as line, s.column as column
         `);

         const symbols: Symbol[] = symbolRows.map(row => ({
           id: String(row.id),
           moduleId: String(row.moduleId),
           name: String(row.name),
           kind: String(row.kind) as Symbol['kind'],
           visibility: String(row.visibility) as Symbol['visibility'],
           location: {
             file: String(row.file),
             line: Number(row.line),
             column: Number(row.column),
           },
         }));

         // 加载依赖关系
         const depRows = await this.queryRows(`
           MATCH (source:Module)-[:IMPORTS]->(target:Module)
           RETURN source.id as sourceId, target.id as targetId
         `);

         const dependencies: Dependency[] = depRows.map(row => ({
           sourceId: String(row.sourceId),
           targetId: String(row.targetId),
           type: 'import' as const,
         }));

         return {
           project: {
             id: meta ? String(meta.projectId) : '',
             name: meta ? String(meta.projectName) : '',
             rootPath: meta ? String(meta.rootPath) : this.projectPath ?? '',
             createdAt: meta ? new Date(String(meta.createdAt)) : new Date(),
             updatedAt: meta ? new Date(String(meta.updatedAt)) : new Date(),
           },
           modules,
           symbols,
           dependencies,
         };
       }
