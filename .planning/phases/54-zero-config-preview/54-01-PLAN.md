export interface DependencyExtractionResult {
  direct: string[];

4. Implement `extractNodeDeps(rootDir: string): string[]`:
   - Read `package.json` via `readFileSync(path.join(rootDir, 'package.json'), 'utf8')`
   - Parse JSON, extract `Object.keys(pkg.dependencies || {})` and `Object.keys(pkg.devDependencies || {})`
   - Return concatenated array. On any error (ENOENT, JSON parse failure), return `[]`
   - Pure function, no `process.exit`

5. Implement `extractGoDeps(rootDir: string): string[]`:
   - Read `go.mod` via `readFileSync`
   - Scan lines for `require (` block and single-line `require` statements
   - Strip `// indirect` and `// direct` inline comments before parsing
   - Extract dependency path (the module name before the version) via regex: `/^(\S+)\s+v\S+/` inside require blocks, `/^require\s+(\S+)\s+v\S+/` for single-line requires
   - Return array of module paths. On any error, return `[]`

6. Implement `extractRustDeps(rootDir: string): Promise<string[]>` (async because smol-toml is ESM):
   - Dynamic import: `const { parse: parseToml } = await import('smol-toml');`
   - Read `Cargo.toml` via `readFileSync`
   - Parse with `parseToml(content)`
   - Extract `Object.keys(parsed.dependencies || {})` and `Object.keys(parsed['dev-dependencies'] || {})`
   - Return concatenated array. On any error (including smol-toml import failure), return `[]`

7. Implement `extractPythonDeps(rootDir: string): Promise<string[]>` (async):
   - Dynamic import smol-toml
   - Read `pyproject.toml` via `readFileSync`
   - Parse with `parseToml(content)`
   - PEP 621 format: extract `(parsed.project as Record<string, unknown>)?.dependencies` — each entry is like `"requests>=2.28"`, extract package name via `/^([A-Za-z0-9_-]+)/`
   - Poetry format: extract `Object.keys(((parsed.tool as Record<string, unknown>)?.poetry as Record<string, unknown>)?.dependencies || {})`
   - Return concatenated array. On any error, return `[]`

8. Implement `extractDependencies(rootDir: string): Promise<DependencyExtractionResult>`:
   - Call all four extractors (node/go sync, rust/python async)
   - Concatenate all results, deduplicate via `Array.from(new Set(...))`
   - Return `{ direct: deduped, count: deduped.length }`

export interface ComplexityHotspot {
  file: string;
  score: number;
  functions: number;
}

3. Define the JS/TS extension allowlist as a constant:
```typescript
const JS_TS_EXTENSIONS = new Set(['.js', '.jsx', '.ts', '.tsx', '.mjs', '.cjs']);
```

4. Implement `scanFileComplexity(filePath: string, rootDir: string): ComplexityHotspot | null`:
   - Check file extension: `const ext = path.extname(filePath); if (!JS_TS_EXTENSIONS.has(ext)) return null;` (mitigates RESEARCH Pitfall 6)
   - Read file via `readFileSync(filePath, 'utf8')`
   - Call `escomplex.analyzeModule(source)` — import as `import escomplex from 'typhonjs-escomplex';`
   - Return `{ file: path.relative(rootDir, filePath), score: report.aggregate.cyclomatic, functions: report.methods.length }`
   - On any error (file read, escomplex parse), return `null` — do NOT throw (per Claude's Discretion: individual file failures should not abort the entire scan)

5. Implement `scanComplexity(files: string[], rootDir: string): ComplexityHotspot[]`:
   - Map files through `scanFileComplexity`
   - Filter out null results: `.filter((h): h is ComplexityHotspot => h !== null)`
   - Sort by score descending: `.sort((a, b) => b.score - a.score)`
   - Take top 5: `.slice(0, 5)` (per D-08)
   - Return the hotspots array
   - Note: This function is synchronous because escomplex is synchronous and readFileSync is synchronous
