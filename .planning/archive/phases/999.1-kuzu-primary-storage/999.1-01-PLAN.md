```typescript
async determineStorageType(config: StorageConfig): Promise<StorageType> {
  if (config.type !== 'auto' && config.type !== 'kuzudb') {
    return config.type;
  }

  // 优先尝试 KùzuDB
  const kuzuAvailable = await this.checkKuzuAvailability();
  if (kuzuAvailable) {
    return 'kuzudb';
  }

  // Fallback 到文件系统
  if (config.type === 'kuzudb') {
    logger.warn('KùzuDB unavailable, falling back to filesystem');
  }
  return 'filesystem';
}

private async checkKuzuAvailability(): Promise<boolean> {
  try {
    const kuzu = await import('kuzu');
    // 尝试创建临时数据库验证功能正常
    const tempDb = new kuzu.Database(':memory:');
    await tempDb.close();
    return true;
  } catch {
    return false;
  }
}
```
