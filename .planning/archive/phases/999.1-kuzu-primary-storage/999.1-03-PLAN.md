// KuzuDBStorage.ts
async migrateFromFilesystem(fsStorage: FileSystemStorage): Promise<boolean> {
  const existingData = await fsStorage.loadCodeGraph();

  if (!existingData || existingData.modules.length === 0) {
    return false; // 无数据需要迁移
  }

  // 检查 KùzuDB 是否已有数据
  const kuzuData = await this.loadCodeGraph();
  if (kuzuData.modules.length > 0) {
    logger.info('KùzuDB already contains data, skipping migration');
    return false;
  }

  // 执行迁移
  await this.saveCodeGraph(existingData);

  // 备份原文件
  await fsStorage.backup();
  logger.info(`Migrated ${existingData.modules.length} modules to KùzuDB`);

  return true;
}
```

## 迁移流程

1. 检测到 `storage.type` 切换到 `kuzudb` 或 `auto` 选择 KùzuDB
2. 检查文件系统中是否存在现有数据
3. 检查 KùzuDB 是否已有数据（避免重复迁移）
4. 如果文件系统有数据且 KùzuDB 为空，执行迁移
5. 迁移成功后保留或备份原文件

    import { rename } from 'fs/promises';

    /**
     * 备份现有数据文件
     * @returns 备份文件路径，如果没有数据文件则返回 null
     */
    async backup(): Promise<string | null> {
      this.ensureInitialized();

      if (!this.dataFilePath) {
        return null;
      }

      try {
        // 检查文件是否存在
        await readFile(this.dataFilePath, 'utf-8');

        // 创建带时间戳的备份文件名
        const timestamp = new Date().toISOString().replace(/[:.]/g, '-');
        const backupPath = `${this.dataFilePath}.backup.${timestamp}`;

        // 重命名文件作为备份
        await rename(this.dataFilePath, backupPath);

        return backupPath;
      } catch (error) {
        if (
          typeof error === 'object' &&
          error !== null &&
          'code' in error &&
          error.code === 'ENOENT'
        ) {
          // 文件不存在，无需备份
          return null;
        }
        throw new StorageError(
          'Failed to backup data file',
          'BACKUP_FAILED',
          error
        );
      }
    }
    ```

       /**
        * 从文件系统存储迁移数据到 KùzuDB
        * @param fsStorage - 文件系统存储实例
        * @returns 是否执行了迁移
        */
       async migrateFromFilesystem(fsStorage: FileSystemStorage): Promise<boolean> {
         this.ensureInitialized();

         // 检查 KùzuDB 是否已有数据
         if (await this.hasExistingData()) {
           console.log('[KuzuDBStorage] KùzuDB already contains data, skipping migration');
           return false;
         }

         // 加载文件系统数据
         let fsData: CodeGraph;
         try {
           fsData = await fsStorage.loadCodeGraph();
         } catch (error) {
           console.log('[KuzuDBStorage] No filesystem data to migrate');
           return false;
         }

         // 检查是否有数据需要迁移
         if (fsData.modules.length === 0) {
           console.log('[KuzuDBStorage] Filesystem data is empty, no migration needed');
           return false;
         }

         console.log(`[KuzuDBStorage] Migrating ${fsData.modules.length} modules from filesystem...`);

         // 执行迁移
         await this.saveCodeGraph(fsData);

         // 备份原文件
         try {
           const backupPath = await fsStorage.backup();
           if (backupPath) {
             console.log(`[KuzuDBStorage] Backed up filesystem data to: ${backupPath}`);
           }
         } catch (backupError) {
           console.warn('[KuzuDBStorage] Failed to backup filesystem data:', backupError);
           // 备份失败不影响迁移成功
         }

         console.log(`[KuzuDBStorage] Successfully migrated ${fsData.modules.length} modules to KùzuDB`);
         return true;
       }

    /**
     * 为项目创建存储实例
     * @param projectPath - 项目路径
     * @param config - 存储配置
     * @returns 初始化的存储实例
     */
    async createForProject(projectPath: string, config: StorageConfig): Promise<IStorage> {
      const storage = await this.create(config);
      await storage.initialize(projectPath);

      // 如果是 KùzuDB 存储，尝试从文件系统迁移数据
      if (storage.type === 'kuzudb') {
        await this.tryMigrateFromFilesystem(storage as KuzuDBStorage, projectPath, config);
      }

      return storage;
    }

    /**
     * 尝试从文件系统迁移数据到 KùzuDB
     */
    private async tryMigrateFromFilesystem(
      kuzuStorage: KuzuDBStorage,
      projectPath: string,
      config: StorageConfig
    ): Promise<void> {
      try {
        // 创建文件系统存储实例用于读取旧数据
        const fsStorage = new FileSystemStorage({
          ...config,
          type: 'filesystem',
        });
        await fsStorage.initialize(projectPath);

        // 执行迁移
        const migrated = await kuzuStorage.migrateFromFilesystem(fsStorage);

        if (migrated) {
          console.log('[StorageFactory] Data migration completed successfully');
        }

        // 关闭文件系统存储
        await fsStorage.close();
      } catch (error) {
        // 迁移失败不影响正常使用，只记录警告
        console.warn('[StorageFactory] Failed to migrate data from filesystem:', error);
      }
    }
